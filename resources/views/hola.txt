       function soloNumeros(e) {
            key = e.keyCode || e.which;
            teclado = String.fromCharCode(key);

            numero = "0123456789";
            especiales = "8-37-38-46";
            tecla_especial = false;
            for (var i in especiales) {
                if (key == especiales[i])
                    tecla_especial = true;
            }
            if (numero.indexOf(teclado) == -1 && !tecla_especial) {
                return false;
            }


        }
